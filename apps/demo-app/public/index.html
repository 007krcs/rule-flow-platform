üìö Quick Start Code

// Install SDK
npm install @ruleflow/sdk

// Use SDK
import { createRuleFlowSDK } from '@ruleflow/sdk';

const sdk = createRuleFlowSDK({
  apiUrl: 'http://localhost:3003'
});

// Execute rules
const result = await sdk.executeRules({
  ruleSetId: 'PECULIARITY_RULES_DE_FR',
  context: {
    userRole: 'ADMIN',
    country: 'DE',
    programId: '123',
    issuerId: 'ISSUER_001'
  },
  data: {
    currency: 'EUR',
    amount: 15000
  }
});

console.log(result.data); // { paymentTerm: 30, ... }
